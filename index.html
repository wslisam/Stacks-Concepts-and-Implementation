stack = [None] * MAX_SIZE
top = -1  # -1 indicates the stack is currently empty

def push(item):
    global top
    # Step 1: Check for Overflow (Is the array full?)
    if top >= MAX_SIZE - 1:
        print("Error: Stack Overflow!")
        return False
    
    # Step 2: Move pointer FIRST
    top += 1
    
    # Step 3: Insert item
    stack[top] = item
    return True

def pop():
    global top
    # Step 1: Check for Underflow (Is the array empty?)
    if top == -1:
        print("Error: Stack Underflow!")
        return None
    
    # Step 2: Retrieve the item
    item = stack[top]
    
    # Step 3: Move pointer DOWN (Logical Deletion)
    top -= 1
    return item

using namespace std;

const int MAX_SIZE = 5;
int stack[MAX_SIZE];
int top = -1; // -1 indicates the stack is empty

bool push(int item) {
    // Step 1: Check for Overflow
    if (top >= MAX_SIZE - 1) {
        cout << "Error: Stack Overflow!" << endl;
        return false;
    }
    
    // Step 2: Increment top pointer FIRST
    top++;
    
    // Step 3: Insert item at the new top position
    stack[top] = item;
    return true;
}

int pop() {
    // Step 1: Check for Underflow
    if (top == -1) {
        cout << "Error: Stack Underflow!" << endl;
        return -1; // -1 indicates error here
    }
    
    // Step 2: Retrieve the item
    int item = stack[top];
    
    // Step 3: Decrement top pointer (Logical Deletion)
    top--;
    return item;
}
#include <iostream>
using namespace std;

const int MAX_SIZE = 5;
int stack[MAX_SIZE];
int top = -1; // -1 代表堆疊目前是空的

bool push(int item) {
    // 步驟 1: 檢查是否溢位 (Overflow)
    if (top >= MAX_SIZE - 1) {
        cout << "錯誤: 堆疊溢位 (Stack Overflow)!" << endl;
        return false;
    }
    
    // 步驟 2: 先將指標遞增 (+1)
    top++;
    
    // 步驟 3: 在新的頂部位置插入資料
    stack[top] = item;
    return true;
}

int pop() {
    // 步驟 1: 檢查是否下溢 (Underflow)
    if (top == -1) {
        cout << "錯誤: 堆疊下溢 (Stack Underflow)!" << endl;
        return -1; // 假設 -1 代表發生錯誤
    }
    
    // 步驟 2: 取得目前最頂端的資料
    int item = stack[top];
    
    // 步驟 3: 指標遞減 (-1) 完成邏輯刪除
    top--;
    return item;
}
